```
# syntax=docker/dockerfile:1.7

ARG NODE_VERSION=22-alpine
ARG PB_VERSION=0.37.1
ARG S6_OVERLAY_VERSION=3.2.2.0
ARG CADDY_VERSION=2.11.2

############################################
# 1. deps
############################################
FROM node:${NODE_VERSION} AS deps
WORKDIR /app
COPY package.json package-lock.json ./
RUN npm ci --no-audit --no-fund

############################################
# 2. builder
############################################
FROM node:${NODE_VERSION} AS builder
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY . .
RUN npm run build

############################################
# 3. runtime
############################################
FROM node:${NODE_VERSION} AS runtime
ARG TARGETARCH
ARG PB_VERSION
ARG S6_OVERLAY_VERSION

RUN apk add --no-cache xz ca-certificates tzdata curl unzip \
    && rm -rf /var/cache/apk/*

# s6-overlay v3 noarch
ADD https://github.com/just-containers/s6-overlay/releases/download/v${S6_OVERLAY_VERSION}/s6-overlay-noarch.tar.xz /tmp/
ADD https://github.com/just-containers/s6-overlay/releases/download/v${S6_OVERLAY_VERSION}/s6-overlay-noarch.tar.xz.sha256 /tmp/
RUN cd /tmp && sha256sum -c s6-overlay-noarch.tar.xz.sha256 \
    && tar -C / -Jxpf s6-overlay-noarch.tar.xz \
    && rm -f s6-overlay-noarch.tar.xz s6-overlay-noarch.tar.xz.sha256

# s6-overlay arch tarball (TARGETARCH -> S6_ARCH mapping)
RUN case "${TARGETARCH}" in \
      amd64) S6_ARCH=x86_64 ;; \
      arm64) S6_ARCH=aarch64 ;; \
      *) echo "Unsupported arch: ${TARGETARCH}" && exit 1 ;; \
    esac \
    && curl -fsSL -o /tmp/s6-arch.tar.xz "https://github.com/just-containers/s6-overlay/releases/download/v${S6_OVERLAY_VERSION}/s6-overlay-${S6_ARCH}.tar.xz" \
    && curl -fsSL -o /tmp/s6-arch.tar.xz.sha256 "https://github.com/just-containers/s6-overlay/releases/download/v${S6_OVERLAY_VERSION}/s6-overlay-${S6_ARCH}.tar.xz.sha256" \
    && cd /tmp && sha256sum -c s6-arch.tar.xz.sha256 \
    && tar -C / -Jxpf s6-arch.tar.xz \
    && rm -f /tmp/s6-arch.tar.xz /tmp/s6-arch.tar.xz.sha256

# Caddy 2.11.2 static binary from official image
COPY --from=caddy:2.11.2-alpine /usr/bin/caddy /usr/bin/caddy

# PocketBase 0.37.1 (checksum-verified)
RUN case "${TARGETARCH}" in \
      amd64) PB_ARCH=amd64 ;; \
      arm64) PB_ARCH=arm64 ;; \
      *) echo "Unsupported arch: ${TARGETARCH}" && exit 1 ;; \
    esac \
    && curl -fsSL -o /tmp/pb.zip "https://github.com/pocketbase/pocketbase/releases/download/v${PB_VERSION}/pocketbase_${PB_VERSION}_linux_${PB_ARCH}.zip" \
    && curl -fsSL -o /tmp/pb.sha "https://github.com/pocketbase/pocketbase/releases/download/v${PB_VERSION}/checksums.txt" \
    && grep "pocketbase_${PB_VERSION}_linux_${PB_ARCH}.zip" /tmp/pb.sha | sha256sum -c - \
    && unzip -q /tmp/pb.zip -d /usr/local/bin/ \
    && chmod +x /usr/local/bin/pocketbase \
    && rm -f /tmp/pb.zip /tmp/pb.sha

# Next.js standalone output (Pitfall #2: explicit COPY of public/ and .next/static/)
WORKDIR /app
COPY --from=builder /app/.next/standalone ./
COPY --from=builder /app/.next/static ./.next/static
COPY --from=builder /app/public ./public

# PB migrations dir (empty in phase 1, phase 2 populates)
RUN mkdir -p /app/pb_migrations

# Runtime configs
COPY docker/Caddyfile /etc/caddy/Caddyfile
COPY docker/s6-rc.d /etc/s6-overlay/s6-rc.d

# Data dir (Pitfall #9: chown to node UID 1000)
RUN mkdir -p /app/data/pb_data \
    && chown -R node:node /app/data

ENV NODE_ENV=production \
    HOSTNAME=127.0.0.1 \
    PORT=3001 \
    S6_KEEP_ENV=1 \
    S6_BEHAVIOUR_IF_STAGE2_FAILS=2 \
    S6_VERBOSITY=1

EXPOSE 3000
VOLUME ["/app/data"]

HEALTHCHECK --interval=30s --timeout=5s --start-period=30s --retries=3 \
  CMD curl -fsS http://127.0.0.1:3000/api/health || exit 1

ENTRYPOINT ["/init"]
```

```sh
#!/bin/sh
set -eu
IMAGE="${1:-homekeep:test}"
BYTES=$(docker inspect "$IMAGE" --format '{{.Size}}')
MB=$((BYTES / 1024 / 1024))
LIMIT=300
echo "Image $IMAGE = ${MB}MB (limit ${LIMIT}MB)"
if [ "$MB" -gt "$LIMIT" ]; then
  echo "FAIL: image exceeds ${LIMIT}MB"
  exit 1
fi
echo "OK"
```

**scripts/check-multiarch.sh** -- validates a pushed multi-arch manifest has both architectures:

```sh
#!/bin/sh
set -eu
IMAGE="${1:-ghcr.io/OWNER/homekeep:latest}"
echo "Inspecting manifest for $IMAGE..."
MANIFEST=$(docker buildx imagetools inspect "$IMAGE" 2>&1)
echo "$MANIFEST"

if ! echo "$MANIFEST" | grep -q 'linux/amd64'; then
  echo "FAIL: manifest missing linux/amd64"
  exit 1
fi
if ! echo "$MANIFEST" | grep -q 'linux/arm64'; then
  echo "FAIL: manifest missing linux/arm64"
  exit 1
fi
echo "OK: manifest contains linux/amd64 and linux/arm64"
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub releases -> Docker image | External binaries (s6-overlay, PocketBase) downloaded and baked in |
| Docker daemon -> built image | Image layers permanent; sensitive data would be extractable |
| Host filesystem -> /app/data volume | UID/GID mismatch can break writes or expose data |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-02-01 | Tampering | s6-overlay tarballs, PocketBase zip, Caddy binary | mitigate | All three are checksum-verified via `sha256sum -c` against publisher-provided `.sha256` / `checksums.txt` files before extraction; Caddy binary is sourced from the signed `caddy:2.11.2-alpine` official image |
| T-01-02-02 | Information Disclosure | Build-time secrets leaked into image layers | mitigate | Dockerfile contains zero secret ARGs; acceptance_criteria greps to enforce. Release workflow (01-06) uses GHCR via `GITHUB_TOKEN` which is never passed as ARG -- login is a separate step |
| T-01-02-03 | Information Disclosure | NEXT_PUBLIC_POCKETBASE_URL baked into bundle | mitigate | Dockerfile never sets NEXT_PUBLIC_*; 01-01 source code uses `window.location.origin`; acceptance_criteria greps for absence in Dockerfile, and 01-06 CI adds a grep check against the built image's static bundle per RESEARCH.md §Pitfall 3 phase-1 action |
| T-01-02-04 | Elevation of Privilege | Container runs as root | accept (phase 1) | s6 runs as PID 1 (root, required); individual services (pocketbase, nextjs) should exec as `node` via s6-setuidgid -- implemented in 01-03 run scripts. Full USER directive switch is deferred pending s6 init testing in phase 1 integration |
| T-01-02-05 | Denial of Service | Image exceeds 300MB on one of the target arches | mitigate | scripts/check-image-size.sh enforces in CI; if size is exceeded, switch alpine base to slim variant or drop unzip after use |
| T-01-02-06 | Tampering | Multi-arch manifest missing an architecture | mitigate | scripts/check-multiarch.sh asserts both linux/amd64 and linux/arm64 in the pushed manifest; release.yml (01-06) runs this after push |
</threat_model>
